Backend


						# Base image
						FROM node:14.17-alpine
						# Set working directory
						WORKDIR /app
						# Copy package.json
						COPY package*.json ./
						# Install dependencies
						RUN npm install
						# Copy source code
						COPY . .
						# Expose port 3000
						EXPOSE 3000
						# Start the application
						CMD ["npm", "start"]

Frontend


						# Base image
						FROM node:14.17-alpine
						# Set working directory
						WORKDIR /app
						# Copy package.json and package-lock.json files
						COPY package*.json ./
						# Install dependencies
						RUN npm install
						# Copy source code
						COPY . .
						# Expose port 5173
						EXPOSE 5173
						# Start the application
						CMD ["npm", "run", "dev"]

docker-compose


					version: '3'

					# Using volumes
					volumes:
						  db-data:

					networks:
					 	 app-network:
							driver: bridge

					services:
						#service for mongo
						db:
							# Pulling the image called "mongo"
							image: mongo
							# restart the container if it is already running
							restart: always
							# defines a volume named db-data
							volumes:
								- db-data:/data/db
							networks:
								- app-network

						backend:
							# fetching the image from my Docker Hub
							image: flyard/todo-backend
							depends_on:
								- db
							# Link to connect to mongo
							environment:
								- MONGO_URI=mongodb://db:27017/
							# Defining ports
							ports:
								- "3000:3000"
							volumes:
								- ./backend:/app/backend
							networks:
								- app-network

						frontend:
							image: flyard/todo-frontend
							depends_on:
								- backend
							ports:
								- "5173:5173"
							volumes:
								- type: bind
								  source: ./frontend/src/components/cardList.jsx
								  target: /app/src/components/cardList.jsx
							networks:
								- app-network
